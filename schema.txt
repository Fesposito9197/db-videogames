db-videogiochi:

VIDEOGIOCATORE: 

- ID : INT, AUTOINCREMENT, NOTNULL, UNIQUE; (PK)
- nome : VARCHAR(100), NOTNULL; 
- cognome: VARCHAR(100), NOTNULL;
- nickname : VARCHAR(50), NOTNULL, UNIQUE;
- città : VARCHAR(100), NOTNULL;

TORNEO:

- ID : INT, AUTOINCREMENT, NOTNULL, UNIQUE; (PK)
- nome : VARCHAR(100), NOTNULL;
- anno : DATE, NOTNULL;
- città_svolgimento : VARCHAR(100), NOTNULL;

SOFTWARE HOUSE:

- ID : INT, AUTOINCREMENT, NOTNULL, UNIQUE; (PK)
- nome : VARCHAR(100), NOTNULL;
- p_iva : VARCHAR(255), NOTNULL, UNIQUE;
- città : VARCHAR(100), NOTNULL;
- nazione : VARCHAR(50), NOTNULL;

VIDEOGAME:

- ID : INT, AUTOINCREMENT, NOTNULL, UNIQUE; (PK)
- nome : VARCHAR(100), NOTNULL;
- trama : TEXT, NOTNULL;
- data_rilascio : DATE, NOTNULL;
- pegi_id : INT, AUTOINCREMENT, NOTNULL, UNIQUE; (FK)
- categoria_id : AUTOINCREMENT, NOTNULL, UNIQUE; (FK)
- software_house_id : AUTOINCREMENT, NOTNULL, UNIQUE; (FK)

PIATTAFORMA:

- ID : INT, AUTOINCREMENT, NOTNULL, UNIQUE; (PK)
- nome : VARCHAR(100), NOTNULL;

CATEGORIA:

- ID : INT, AUTOINCREMENT, NOTNULL, UNIQUE; (PK)
- nome : VARCHAR(100), NOTNULL;

PEGI:

- ID : INT, AUTOINCREMENT, NOTNULL, UNIQUE; (PK)
- nome : VARCHAR(100), NOTNULL;

RECENSIONI:

- ID : INT, AUTOINCREMENT, NOTNULL, UNIQUE; (PK)
- titolo : VARCHAR(100), NOTNULL;
- testo : TEXT, NOTNULL;
- rating : INT, NOTNULL;
- videogiocatore_id : AUTOINCREMENT, NOTNULL, UNIQUE; (FK)
- videogame_id : AUTOINCREMENT, NOTNULL, UNIQUE; (FK)
